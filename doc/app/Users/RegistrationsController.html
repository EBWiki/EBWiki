def create
  if gotcha_valid?
    super
  else
    clean_up_passwords resource
    @validatable = devise_mapping.validatable?
    if @validatable
      @minimum_password_length = resource_class.password_length.min
    end
    flash[:notice] = 'Invalid Captcha'
    redirect_to "/users/sign_up"
  end
end
